rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    function isAuthed() { return request.auth != null; }
    function isRoom(id) { return exists(/databases/$(database)/documents/rooms/$(id)); }

    match /rooms/{roomId} {
      allow read: if true;
      allow create: if isAuthed();
      allow update: if isAuthed();
      // TTL用に expiresAt を許可（必要であれば制限を精緻化）
      allow delete: if false;

      match /players/{playerId} {
        allow read: if true;
        allow create: if isAuthed() && request.auth.uid == playerId;
        allow update: if isAuthed() && request.auth.uid == playerId;
        allow delete: if isAuthed() && request.auth.uid == playerId;
      }

      match /chat/{msgId} {
        allow read: if true;
        allow create: if isAuthed();
        allow delete, update: if false;
      }
    }
  }
}
